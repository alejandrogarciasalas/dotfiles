[alias]
	# one-letter shortcuts
	a = add
	b = branch
	c = commit
	d = diff
	f = fetch
	l = log
	m = merge
	o = checkout
	p = pull
	s = status

	# add
	aa = add --all
	ap = add --patch
	au = add --update

	# branch
	bm = branch --merged
	bnm = branch --no-merged

	# commit
	ca = commit --amend
	cane = commit --amend --no-edit
	cm = commit --message

	# checkout
	co = checkout

	# cherry-pick
	cp = cherry-pick
	cpa = cherry-pick --abort
	cpc = cherry-pick --continue

	# diff
	dc = diff --cached
	ds = diff --staged
	dw = diff --word-diff

	# log
	lg = log --graph
	lo = log --oneline
	lp = log --patch
	ll = log --graph --topo-order --date=short --abbrev-commit --decorate --all --boundary --pretty=format:'%Cgreen%ad %Cred%h%Creset -%C(yellow)%d%Creset %s %Cblue[%cn]%Creset'

	# pull
	pf = pull --ff-only
	pr = pull --rebase

	# rebase
	rb = rebase
	rba = rebase --abort
	rbc = rebase --continue
	rbs = rebase --skip
	rbi = rebase --interactive @{upstream}

	# status
	ss = status --short
	ssb = status --short --branch

	# reset/undo
	uncommit = reset --soft HEAD~1
	unstage = reset HEAD

	# stash
	save = stash push
	pop = stash pop

	# push/pull current branch
	push1 = "!git push origin $(git branch-name)"
	pull1 = "!git pull origin $(git branch-name)"
	publish = "!git push --set-upstream origin $(git branch-name)"

	# cleanup
	cleanest = clean -ffdx
	prune-all = !git remote | xargs -n 1 git remote prune

	# wip - save/restore work-in-progress
	wip = !"git add --all; git ls-files --deleted -z | xargs -0 git rm; git commit --message=wip"
	unwip = !"git log -n 1 | grep -q -c wip && git reset HEAD~1"

	# merge helpers
	ours = "!f() { git checkout --ours $@ && git add $@; }; f"
	theirs = "!f() { git checkout --theirs $@ && git add $@; }; f"
